package main

import (
	"fmt"
	elasticsearch7 "github.com/elastic/go-elasticsearch/v7"
	"log"
	"testing"
)

func TestElasticsearch1(t *testing.T) {
	cfg := elasticsearch7.Config{
		Addresses: []string{
			"http://127.0.0.1:9200",
		},
	}
	es, err := elasticsearch7.NewClient(cfg)
	if err != nil {
		log.Fatal(err)
	}

	res, err := es.Info()
	if err != nil {
		log.Println(err)
	}
	defer res.Body.Close()

	log.Println(res)
}

func TestAtomic(t *testing.T) {
	v := uint(3)
	a := uint(0) + v
	b := ^uint(0) + v
	fmt.Printf("%d, %b\n", a, a)
	fmt.Printf("%d, %b\n", b, b)
}
